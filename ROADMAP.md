#### Core Data Schema
```swift
// Recipe Entity
Recipe {
    id: UUID
    title: String
    ingredients: String
    instructions: String
    category: String
    difficulty: Int16
    estimatedTime: Int16
    cookingRecords: [CookingRecord]
    createdAt: Date
    updatedAt: Date
}

// CookingRecord Entity  
CookingRecord {
    id: UUID
    recipe: Recipe
    actualTime: Int16
    photos: [PhotoData]
    notes: String
    rating: Int16
    experienceGained: Int16
    completedAt: Date
}
```
